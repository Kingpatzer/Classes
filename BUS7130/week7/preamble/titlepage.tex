\title{Accenture PLC Strategy Optimization}
\shorttitle{OPTIMIZING STRATEGY}

\author{David A. Wagle}
\affiliation{North Central University}

\leftheader{Wagle}

\abstract{This paper attempts to optimize the strategic plan that has been so far developed. In this effort, intelligence and inforamation gathering will be reviewed, as will the initial strategy formulation. A master project plan will be proposed, with implementation guideposts and details around monitoring the implementation, reviewing the effort, and updating the plan as needed.}

\keywords{Accenture, Strategic Planning}
