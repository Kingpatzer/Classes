\title{Selection of an Appropriate Strategy for Accenture}
\shorttitle{STRATEGY SELECTION}

\author{David A. Wagle}
\affiliation{North Central University}

\leftheader{Wagle}

\abstract{This paper examines the selection of an appropriate strategy for Accenture from the four central concepts outlined by Tracy in his text. This paper specifically considers how each of these strategies aligns to Accenture's mission, vision and values.}
