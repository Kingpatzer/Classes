\begin{frame}
  \frametitle{Managing Innovation}
  \framesubtitle{A Changing Challenge}
  \begin{itemize}
    \item<1-> Just as the term has shifted over time, so too has how firms think of Innovation
          \begin{\itemize}
            \item<1-> R\&D Labs
            \item<1-> Meeting Customer Expectations
            \item<1-> Enhancing Firm Efficiency
            \item<1-> Enhancing Supply-Chain / Partnership Efficiency
          \end{itemize}

          \note[item]<1-> {\scriptsize{Innovation moved from a linear process of research and development where an invention was produced then turned into a product idea, then a product, and finally introduced into production to a broader concept of developing synergies and efficiencies across multiple product lines and eventually multiple companies \parencite{orttEvolutionInnovationManagement2008}. }}


    \item<2-> No Consensus on ``How to manage''


          \note[item]<2-> {\scriptsize{\textcite{tiddInnovationManagementContext2001} argues that context impacts innovation management, and there may not be a single model that can be applied successfully across companies. }}

  \end{itemize}
\end{frame}
