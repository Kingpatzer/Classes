\begin{frame}
  \frametitle{Innovation's Value}
  \framesubtitle{A Myth? }
  \begin{itemize}
    \item<1-> For any one company, Innovation can not matter

          \note[item]<1-> {\scriptsize{\textcite{drakemanRiskDeRiskingInnovation2020} show that innovative companies fail more often than they succeed in some industries. However, the companies that succeed must be innovative, at least in some industries.}}

    \item<2-> Innovation matters over industry sectors

          \note[item]<2-> {\scriptsize{The real impact here is that this research shows that industry sectors that are innovative are successful in terms of returns, but individual companies mostly fail! This means that innovation is an aggregate benefit but a specific risk.}}

  \end{itemize}
\end{frame}
