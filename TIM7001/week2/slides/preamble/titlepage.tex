\title{SWOT Analysis: Accenture PLC}
\shorttitle{SWOT ACCENTURE}

\author{David A. Wagle}
\affiliation{North Central University}

\leftheader{Wagle}

\abstract{This paper is a critical examination of the strengths, weaknesses, opportunities, and threats faced by Accenture PLC in 2020. Of particular interest is in the impact of COVID-19 on world-wide commerce, the associated economic slowdown, and the impact of this slowdown on the short-term viability of corporate consulting as a specialty field. }

\keywords{Accenture, SWOT, Consulting}
