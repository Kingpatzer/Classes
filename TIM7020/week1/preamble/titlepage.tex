\title{TIM7020 Week 1: Annotated Bibliography}
\shorttitle{ANNOTATED BIB.}

\author{David A. Wagle}
\affiliation{Northcentral University}

\leftheader{Wagle}


\abstract{An annotated bibliography covering 10 articles selected with a search query of the terms: Business Intelligence and Analytics; Data Management; and, Empirical or Research or Study. All articles are from the time frame of 2018 until 2021 and the contents are arranged first by calendar year then by primary author last name. This work was completed for the course TIM7020 at Northcentral University as part of the PhD in Business Administration program with a specialization in Information Technology.}

\keywords{Annotated Bibliography, Business Intelligence and Analytics, Data Management}
